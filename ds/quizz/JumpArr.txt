#define CANT_JUMP 0
#define ABLE_TO_JUMP 1

/*naive:*/
ssize_t IsAbleToJumpToEnd(int* arr, size_t size, ssize_t cur_index)
{
	assert(NULL != arr);
	
	if (0 > cur_index || cur_index >= size)
	{
		return CANT_JUMP;
	}
	
	if (cur_index == size - 1)
	{
		return ABLE_TO_JUMP;
	}
	
	return (IsAbleToJumpToEnd(arr, size, cur_index + arr[cur_index]) || IsAbleToJumpToEnd(arr, size, cur_index - arr[cur_index])); 
}

int IsAbleToJumpToEndV2(int* arr, size_t size, ssize_t cur_index)
{
	ssize_t tmp = 0;	
	int result = 0;
	
	assert(NULL != arr);
	
	if (0 > cur_index || cur_index >= (ssize_t)size || 0 == arr[cur_index])
	{
		return CANT_JUMP;
	}
	
	if (cur_index == (ssize_t)size - 1)
	{
		return ABLE_TO_JUMP;
	}
	
	tmp = arr[cur_index];
	arr[cur_index] = 0;
	
	 result = (IsAbleToJumpToEndV2(arr, size, cur_index + tmp) || IsAbleToJumpToEndV2(arr, size, cur_index - tmp));
	 
	 arr[cur_index] = tmp;
	 
	 return result;
}
