int CalcAmountOfRoomsNeed(meet_t* arr, size_t size)
{
	int made_change = 0;
	size_t arr_index = 0;
	size_t rooms_index = 0;
	size_t i = 0;
	meet_t* rooms_arr = NULL;
	
	assert (NULL != arr);
	
	rooms_arr = (meet_t*)malloc(sizeof(meet_t) * size);
	if (NULL == rooms_arr)
	{
		return -1;
	}
	if (2 > size)
	{
		return size;
	}
	
	QuickSortByStartTime(arr);
	rooms_arr[rooms_index++] = arr[arr_index++];
	for (; arr_index < size; ++arr_index)
	{
		made_change = 0;
		for (i = 0; i < rooms_index; ++i)
		{
			if (arr[arr_index].start > rooms_arr[i].end)
			{
				rooms_arr[i] = arr[arr_index];
				i = rooms_index;
				++made_change;
			}
		}
		
		if (!made_change)
		{
			rooms_arr[rooms_index++] = arr[arr_index];
		}
	}
	
	free(rooms_arr); rooms_arr = NULL;
	return rooms_index;
}
