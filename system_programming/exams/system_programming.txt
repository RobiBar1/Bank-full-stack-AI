18. 
א. mutex - מנגנון נעילה לעבודה מקבילית נועד לאפשר יצירת קטע קריטי בקוד שבו רק תהליך/חוט אחד יוכלו לגשת בזמן נתון, למיוטקס יש בעלים ומנעול, שרק מי שנעל אותו יכול לפתוח אותו ובכך הוא מונע מכל תהליך/חוט אחר לגשת לאיזור הזה עד שאותו אחד שנעל אותו יסיים את הקטע קוד הקריטי וישחרר את המנעול.
הוא נועד להגן על משאבים משותפים ל2 חוטים/תהליכים או יותר ובככה למנוע RACE CONDETION וגם בזמן ההמתנה הוא יגרום לתהליך/חוט הלרוונטי לצרוך 0% CPU.

ב. semaphore - מנגנון סנכרון של עבודה מקבילית שנועד לתזמן עבודה בין תהליכים/חוטים שונים אם נרצה להבטיח סדר הגעה מסויים. נוכל לשים אותו בשורה מסויימת בקוד שנרצה לוודא שתהליך מסויים עובר אותה רק אחרי שמשהוא קרה בתהליך אחר ובזמן ההמתנה הוא יגרום לתהליך/חוט הלרוונטי לצרוך 0% CPU.
וזה יעבוד כך: בתוכו יהיה מונה מסויים ורק כאשר הוא יהיה גדול מ0 תהליכים/חוטים אחרים יוכלו להמשיך לעבוד ולעבור את השורה הזאת ששמתי בקוד.

ג. condition varible - מנגנון איתות שיעבוד בתוכו עם MUTEX, ובעצם נשתמש בו לגרום ל2+ תהליכים/חוטים לחכות בלי לצרוך זמן CPU בשורה מסויימת בקוד עד שמשהוא קריטי מבחינתי יקרה בתהליך/חוט אחר ואז שהוא קרה אני האותת לכולם ביחד שאפשר להמשיך ואז הראשון מהם שיקבל זמן CPU(אני לא יכול לדעת מי בדיוק כי אני לא שולט במתזמן של הOS)ינעל את המיוטקס ויכנס לקטע קוד קריטי כאשר כל השאר שיקבלו CPU אחריו יחזרו לחכות עד האיתות הבא וכן הלאה..

ד. critical section - קטע קוד שיכיל איזה שהוא משאב משותף שרק תהליך/חוט אחד יוכל לגשת אליו ולהיות בתוכו בכל פעם ומובטח לי שאם הוא קורה גם אם תהליך/חוט אחר יקבל פתאום זמן CPU לפני שסיימתי אותו עדיין אותם תהליכים/חוטים בוודאות לא יוכלו לאותו משאב משותף.

ה. spin-lock: בעצם יעבוד כמו מיוטקס רק ההבדל הוא שכאן בזמן שהתהליך/חוט רלוונטים ימתינו הם יוכלו לקבל זמן CPU ולבזבז אותו עד שמערכת ההפעלה תקח אותו מהם.

ו. atomic operation - פעולה שמובטח לי שאם היא התחילה היא תסתיים ב100%(כמובן בהנחה שהמחשב לא קורס פתאום ואז היא לא תספיק לסיים את כולה..) בלי שמערכת ההפעלה תקטע אותה ותעביר זמן CPU למשיהוא אחר.

ז. asynchronous operation: פעולה שמתבצעת ולא חוסמת את ביצוע המשך שורות הקוד שבאות אחריה בזמן שהיא ממתינה לתשובה או למשהוא שיתבצע.

נ. reentrancy: הבטחה שאוכל לגשת שוב לאותו קטע קוד(פונקצייה\חוט\תהליך) גם אם הייתי באמצע שלו ואז הפעילו אותי מחדש והכל יעבוד כמו שצריך. צריך בשביל זה כמה תנאים:
- היא לא ניגשת למשאבים משותפים בכלל או שאם היא ניגשת אז אותו משאב נעול בקטע קריטי או שהפעולה עליו תיהיה אטומית
היא לא תשתמש בI/O.

* אז  reentrancy funcion: פונקציה שקמיימת שגם אם בשורה מסויימת שלה היא תקרא לעצמה עדיין הכל יעבוד כמו שצריך.

* אז reentrancy synchronization: מקיים את זה בכך שנגיד הפונקציה מסויימת שתיהיה בת 10 שורות 1 - 10 התחילה לרוץ והיא מבטיחה שגם אם בשורה מסויימת בין 2 - 9 פתאום חוט אחר יקבל זמן CPU ויריץ את אותה פונקציה אז עדיין היא תעבוד כמו שצריך.

ס. priority inversion: מתאר מקרה כזה:
3 תהליכים רצים:
- תהליך א עם עדיפות 1(הכי גבוהה).
- תהליך ב עם עדיפות 2(בינונית).
- תהליך ג עם עדיפות 3(הכי נמוכה).
ואז תהליך ג מקבל זמן CPU ומשתמש במשאב מסויים נגיד מצלמה של מחשב, ואז תוך כדי שהוא משתמש בו נוצר תהליך א OS מחליט לתת עכשיו זמן CPU לתהליך א כי הוא בתעדוף גבוהה יותר אך בתהליך א הוא גם צריך שימוש במצלמה אך היא כרגע בשימוש אז הוא ממתין ואומר לCPU תריץ דברים אחרים עד שהמצלמה תשוחרר, עכשיו נוצר תהליך ב הOS מחליט לתת לו זמן CPU כי הוא בעדיפות גבוהה יותר מתהליך ג ואז תהליך ג לא מקבל הזמדנות לשחרר את המשאב שתהליך א עם העדיפות הגבוהה ביותר מחכה לו ואז נגיד שעכשיו מלא תהליכים עם עדיפות 2 נוצרים ומקבלים זמן CPU על חשבון תהליך ג כי הוא עדיפות נמוכה יותר אז בפועל תהליכים בעדיפות 2 תוקעים את תהליך א עם עדיפות גבוהה יותר לרוץ.

#define SIZE_OF_ARR 64 /* most be pow of 2 for bit trick*/

sem_t sema_can_read;
sem_t sema_can_write;

mutex_t mutex_can_read;
mutex_t mutex_can_write;

static sig_atomic volincte can_read;
static sig_atomic volincte can_write;

typedef struct shared_date_write
{
	size_t write_index;
	int* arr;
	int msg;
}SharedWrite_t;

typedef struct shared_date_read
{
	size_t read_index;
	int* arr;
}SharedRead_t;

void* Producer(void* shared)
{
	SharedWrite_t* shared = (SharedWrite_t*)shared;
	
	assert(NULL != shared);
	
	while(1)
	{
		sem_wait(&sema_can_write);
		
		TryLock(&mutex_can_write);
		
		shared->arr[shared->write_index] = shared->msg;
		shared->write_index = (shared->write_index + 1) & (SIZE_OF_ARR - 1);
		++(shared->msg);
		
		UnLock(&mutex_can_write);
		
		sem_post(&sema_can_read);
		sleep(1);
	}
}


void* Consumer(void* shared)
{
	SharedRead_t* shared = (SharedRead_t*)shared;
	int msg = 0;
	
	assert(NULL != shared);
	
	while(1)
	{
		sem_wait(&sema_can_read);
		
		TryLock(&mutex_can_read);
		
		msg = shared->arr[shared->read_index];
		shared->read_index = (shared->read_index + 1) & (SIZE_OF_ARR - 1);
		
		UnLock(&mutex_can_read);
		
		sem_post(&sema_can_write);
		
		/* DoAction(msg) */
	}
}

#define NUM_READERS 10
#define NUM_WRITERS 20

int main()
{
	p_thread writers[NUM_WRITERS] = {0};
	p_thread writers[NUM_READERS] = {0};
	SharedWrite_t write = {0};
	SharedRead_t read = {0};
	
	write.arr = (int*)calloc(SIZE_OF_ARR, sizeof(int));
	if (NULL == write.arr)
	{
		perror("alloc failed");
		
		return;
	}
	
	write.write_index = 1;
	read.read_index = 0;
	read.arr = write.arr;
	
	Init_sem(&sema_can_read, 0);
	Init_sem(&sema_can_write, SIZE_OF_ARR - 1);

	Init_mutex(&mutex_can_read, &can_read ,0);
	Init_mutex(&mutex_can_write, &can_write ,1);
	
	for (; i < NUM_WRITERS; ++i)
	{
		p_thread_create(&writers[i], NULL, NULL, Consumer, &write);
	}
	
	for (i = 0; i < NUM_READERS; ++i)
	{
		p_thread_create(&readers[i], NULL, NULL, Producer, &read);
	}
	
	for (i = 0; i < NUM_WRITERS; ++i)
	{
		p_thread_join(&writers[i]);
	}
	
	for (i = 0; i < NUM_READERS; ++i)
	{
		p_thread_join(&readers[i]);
	}	
	
	free(write.arr);
	write.arr = NULL;
	
	return 0;
}

34. נקראים גם IPC:

א. shared-memory: אשתמש כאשר הדגש שלי הוא על מהירות העברת המידע בין 2 תהליכים שונים על אותו מחשב ולא חשוב לי לשמור את המידע המשותף בסוף העבודה על הדיסק.

ב. shared-file-memory: אשתמש כאשר ארצה לעבוד מהר וכן חשוב לי לשמור את המידע המשותף בסוף העבודה על הדיסק.

ג. pipe: אשתמש אם ארצה תקשורת חד חיוונית שתהליך A יוכל להעביר מידע להתליך B ותהליך B רק יוכל לקרוא את המידע.

ד. socket: אשתמש כאשר ארצה להעביר מידע בין מחשבים מרוחקים.

ה. signals: אשתמש אם ארצה רק לאותת בצורה מהירה בלי להעביר הרבה מידע בין תהליכים שונים על אותו מחשב.

35. 
א. TCP: חיבור איטי יותר אך נחשב בטוח יותר להעברת וקבלת מידע.
ב. UDP: חיבור מהיר יותר אך נחשב פחות בטוח מTCP להעברת וקבלת מידע.

היתרון של TCP: ווידוא פתיחת קשר עם הצד השני, הגעת הפאקטות לצד השני, קריאה בסדר הנכון של הפאקטות, יכול לדעת שעבר עם או ללא שגיאות.

היתרון של UDP: מאוד מהיר ביחס לTCP, יותר פשוט למימוש(פחות שלבים).

החסרונות של שניהם הם היתרון של השני.

36. 
א. איך להמנע מדליפות זכרון:
עקרון RAII.
- בC: על כל הקצאה(MALLOC\CALLOC) דינאמית או שימוש במשאב(OPEN_FILE) נרצה לעשות אותו בתחילת השימוש עצמו ולא לפני. ואת השחרור שלו נרצה לעשות ישר אחרי שנסיים להשתמש בו בפעם האחרונה. בנוסף נרצה שהאחריות של מי שהקצה/פתח את המשאב תיהיה גם לשחרר אותו.

- בC++ נרצה לעטוף במחלקה ואז אוטומטית הקומפיילר יקרא לCTOR בעת היצירה שלה ולDTOR בעת ההריסה שלה ששם כמובן נשים את ההקצאות והשיחרורים.

ב. איך להמנע מSEG_FUALT - בC וC++ כאשר ניגש לפוינטר נבדוק שהוא מצביע לאן שהוא ולא NULL לפני שנעבוד עליו. אם נפנה לאינדקס מסויים נוודא שהוא בטווח הנכון לפני שנפנה אליו. בC++ נעביר רפרנס אם אפשר וככה זה בטוח לא יהיה NULL ונוכל לחסוך בדיקה ועדיין להיות בטוחים שזה מקום קיים בזכרון.

ג. בC וC++ אחרי שחרור אז נשווה את הפוינטר לNULL וככה נדע שמבחינתי המקום הזה לא וולידי יותר ואם בטעות אנסה לגשת אליו ממקום אחר שם אני אבדוק אם הוא NULL(כמו שציינתי בסעיף א) ולא אגש אליו אם אכן NULL(שחררתי אותו..).
