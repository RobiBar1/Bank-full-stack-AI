namespace ilrd
{

<typename T>
class LRU<T,T>
{
public:
	LRU<>(std::size_t capacaity);
	
	void Put(T key, T value);
	T Get(T key);

private:
	std::size_t m_cap;
	Map<T,T> m_map;
}

<typename T>
LRU<T,T>::LRU(std::size_t capacaity) : m_cap(capacaity), m_map(new Map<T,T>(capacaity))
{
}

<typename T>
LRU<T,T>::Put(T key, T value)
{
	if (!m_map.Find())
	{
		m_map.Add(key, value);
	}
}

<typename T>
T LRU<T,T>::Get(T key)
{
	return m_map.Get(key);
}

} // end ilrd
