int* sumArrInMatrix(int** matrix, size_t row, size_t col)
{
	assert(NULL != matrix);
	assert(NULL != *matrix);
	
	int* res_arr = NULL;
	res_arr = (int*)calloc(row, sizeof(int));
	if (NULL == res_arr)
	{
		return NULL;
	}
	
	for (size_t i = 0; i < row; ++i)
	{
		for (size_t j = 0; j < col; ++j)
		{
			res_arr[i] += matrix[i][j];
		}	
	}
	
	return res_arr;
}
